155 template <int xDims, int yDims, int uDims>
156 void Kalman<xDims, yDims, uDims>::Predict(
157  const Eigen::Matrix<float, uDims, 1>& u) {
158  x_ = F_ * x_ + B_ * u;
159  P_ = F_ * P_ * F_.transpose() + Q_;
160 }
161 
162 template <int xDims, int yDims, int uDims>
163 void Kalman<xDims, yDims, uDims>::Update(
164  const Eigen::Matrix<float, yDims, 1>& y) {
165  Eigen::Matrix<float, yDims, yDims> S = H_ * P_ * H_.transpose() + R_;
166  Eigen::Matrix<float, xDims, yDims> K = P_ * H_.transpose() * S.inverse();
167  x_ = x_ + K * (y - H_ * x_);
168  P_ = (Eigen::Matrix<float, xDims, xDims>::Identity(xDims, xDims) - K * H_) * P_;
169 }
